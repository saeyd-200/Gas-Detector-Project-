int gasSensorA = A0;   
int gasSensorD = 2;    
int buzzer     = 8;   

int gasValue;         
int threshold = 400;   

void setup() {
  pinMode(gasSensorD, INPUT);  
  pinMode(buzzer, OUTPUT);     
  Serial.begin(9600);
}

void loop() {
 gasValue = analogRead(gasSensorA);
  Serial.print("Analog Gas Value: ");
  Serial.println(gasValue);
  int gasState = digitalRead(gasSensorD);

  if (gasValue > threshold || gasState == LOW) { 
    digitalWrite(buzzer, HIGH);   
    Serial.println("Gas Detected!");
  } else {
    digitalWrite(buzzer, LOW);  
    Serial.println("No Gas");
  }

  Serial.println("--------------------");
  delay(500);
}
